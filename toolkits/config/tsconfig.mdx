## Quick Start

<Steps>
  <Step title="Create Project tsconfig.json">
    Create a `tsconfig.json` file in your project root:

    ```json tsconfig.json
    {
      "$schema": "https://json.schemastore.org/tsconfig.json",
      "extends": "@pelatform/tsconfig/base"
    }
    ```
  </Step>

  <Step title="Add Scripts to package.json">
    Add build scripts to your `package.json`:

    ```json package.json
    {
      "scripts": {
        "build": "tsc",
        "build:watch": "tsc --watch",
        "type-check": "tsc --noEmit"
      }
    }
    ```
  </Step>

  <Step title="Configure Your Project">
    Customize the configuration for your project:

    ```json tsconfig.json
    {
      "$schema": "https://json.schemastore.org/tsconfig.json",
      "extends": "@pelatform/tsconfig/base",
      "compilerOptions": {
        "rootDir": "src",
        "outDir": "dist"
      },
      "include": ["src/**/*"],
      "exclude": ["node_modules", "dist"]
    }
    ```

    <Tip>
      Use schema validation for autocomplete and error checking in your editor.
    </Tip>
  </Step>
</Steps>

## Examples

### React Application

```json tsconfig.json
{
  "$schema": "https://json.schemastore.org/tsconfig.json",
  "extends": "@pelatform/tsconfig/react",
  "compilerOptions": {
    "baseUrl": ".",
    "paths": {
      "@/*": ["./src/*"],
      "@/components/*": ["./src/components/*"],
      "@/utils/*": ["./src/utils/*"]
    }
  },
  "include": [
    "src/**/*",
    "vite-env.d.ts"
  ],
  "exclude": [
    "node_modules",
    "dist"
  ]
}
```

## Advanced Configuration

<Accordion title="Custom Path Mapping">
  Configure custom path aliases for cleaner imports:

  ```json tsconfig.json
  {
    "extends": "@pelatform/tsconfig/base",
    "compilerOptions": {
      "baseUrl": ".",
      "paths": {
        "@/*": ["./src/*"],
        "@/components/*": ["./src/components/*"],
        "@/lib/*": ["./src/lib/*"],
        "@/types/*": ["./src/types/*"],
        "@/utils/*": ["./src/utils/*"],
        "@/hooks/*": ["./src/hooks/*"]
      }
    }
  }
  ```
</Accordion>

<Accordion title="Multiple Environments">
  Use different configurations for development and production:

  ```json tsconfig.build.json
  {
    "extends": "./tsconfig.json",
    "compilerOptions": {
      "noEmit": false,
      "declaration": true,
      "declarationMap": true,
      "sourceMap": true
    },
    "include": ["src/**/*"],
    "exclude": ["**/*.test.*", "**/*.spec.*"]
  }
  ```

  ```json tsconfig.test.json
  {
    "extends": "./tsconfig.json",
    "compilerOptions": {
      "noEmit": true,
      "types": ["jest", "node"]
    },
    "include": [
      "src/**/*",
      "**/*.test.*",
      "**/*.spec.*"
    ]
  }
  ```
</Accordion>
