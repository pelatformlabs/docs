  <Tab title="Link Constructor">
    ```typescript link-constructor.ts
    import { linkConstructor } from '@pelatform/utils';

    // Construct URL with options
    const url = linkConstructor({
      domain: 'example.com',
      key: 'page',
      searchParams: { q: 'search' }
    });
    // "https://example.com/page?q=search"

    // Pretty URL (no protocol)
    const pretty = linkConstructor({
      domain: 'example.com',
      pretty: true
    });
    // "example.com"
    ```

    } from '@pelatform/utils';

    });
