    ```typescript stable-sort.ts
    import { stableSort } from '@pelatform/utils';

    const users = [
      { name: 'Alice', role: 'admin', id: 1 },
      { name: 'Bob', role: 'user', id: 2 },
      { name: 'Charlie', role: 'user', id: 3 }
    ];

    // Sort by role, preserving original order for same roles
    const sorted = stableSort(users, (a, b) => a.role.localeCompare(b.role));
    ```
