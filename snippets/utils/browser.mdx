    ```typescript cookies.ts
    import {
      setCookie,
      getCookie,
      removeCookie,
      hasCookie,
      getAllCookies,
      clearAllCookies
    } from '@pelatform/utils';

    interface CookieOptions {
      expires?: number; // Expiration time in days
      path?: string;
      domain?: string;
      secure?: boolean; // HTTPS only
      sameSite?: "strict" | "lax" | "none";
      httpOnly?: boolean; // Server-side only
    }

    // Set a cookie
    setCookie('theme', 'dark', { expires: 30 });

    // Get a cookie
    const theme = getCookie('theme'); // "dark"

    // Check if cookie exists
    const hasTheme = hasCookie('theme'); // true
    ```
